function generateImage(word) { // Create a new canvas element var canvas = document.createElement('canvas'); canvas.width = 250; canvas.height = 250; // Get the 2D rendering context var ctx = canvas.getContext('2d'); // Set background color ctx.fillStyle = '#ffffff'; // You can change this to any color ctx.fillRect(0, 0, canvas.width, canvas.height); // Set text properties ctx.font = '20px Arial'; ctx.fillStyle = '#000000'; // You can change this to any color ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; // Position the text in the center var x = canvas.width / 2; var y = canvas.height / 2; // Draw the text on the canvas ctx.fillText(word, x, y); // Create an image from the canvas var dataURL = canvas.toDataURL('image/png'); // You can use the dataURL or convert it to a blob and create a downloadable link // For simplicity, let's just log the dataURL for now console.log(dataURL); } // Example usage: generateImage('Hello');
